      const findEvenOdd = (N, Arr) => 
      {
         let even=0;
         let odd=0;
         for (let i=0;i<N;i++) {
           if(Arr[i]%2===0) {
             even+=Arr[i];
           }
           else {
             odd+=Arr[i];
           }
         }
         let B=[even,odd]
         return B;
      };

Q7-Unique Color Shirt-

      function Unique_Shirts (arr,N) {
   
        let freq=[];
        for(let i=0;i<1001;i++) {
          freq[i]=0;
        }
        for(let i=0;i<N;i++) {
          freq[arr[i]]++;
        }
        let ans=0;
        for(let i=0;i<1001;i++) {
          if(freq[i]===1) {
            ans++;
          }
        }
        return ans;
      
     }
    
Q8-Birthday Game-

      function Birthday_Game(arr,D ,M) {
        let N= arr.length;
        let count=0;
        for (let i=0;i<N-M+1;i++){
          let sum=0;
          for(let j=i;j<M+i;j++) {
            sum+=arr[j];
          }
          if(sum===D)
          count++;
        }
        return count;
      }
    
Q9-Min and Max-

      function arrayMin(arr) {
        let min=arr[0];
        for(let i=0;i<arr.length;i++) {
          if(arr[i] < min) {
            min=arr[i];
          }
        }
        return min;  
       }
       
       function arrayMax(arr) {
        let max=arr[0];
        for(let i=0;i<arr.length;i++){
          if(arr[i] > max) {
            max=arr[i];
          }
        }
        return max;
